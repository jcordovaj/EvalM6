def index(request):
    if request.user.is_authenticated:
        # Obtener la lista de URLs permitidos para el usuario
        allowed_urls = get_allowed_urls(request.user)
    else:
        allowed_urls = ['/']  # Solo permitir el acceso a la página de inicio
    return render(request, 'index.html', {'allowed_urls': allowed_urls})
